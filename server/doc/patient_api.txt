number: 1
description: Get info -> Called
route: http://localhost:5000/api/patient/info
method: GET
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
returns: {
  _id: String (Database ID),
  id_number: String,
  name: String,
  dob: String,
  address: String,
  status: String,
  current_facility: String,
}

number: 2
description: Get logs -> Called
route: http://localhost:5000/api/patient/logs
method: GET
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
returns: [
  {
    _id: String (Database ID),
    account: String (_id of account),
    action: String,
    time: String (yyyy-mm-dd hh:mm:ss)
  }
]

number: 3
description: Change password for client -> Called
route: http://localhost:5000/api/patient/password
method: PUT
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
body: {
  old_password: String,
  new_password: {
    type: String
    minlength: 6,
    required: true
  }
}
returns: {
  message: String
}

number: 4
description: Buy a package
route: http://localhost:5000/api/patient/packages/id=:id
params: Database ID of package
method: POST
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
body: {
  products: [
    {
      id: String (Database ID of product),
      quantity: Number
    }
  ]
}
returns: {
  message: String
}

number: 5
description: Link account to payment system -> Error
route: http://localhost:5000/api/patient/link
method: PUT
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
returns: {
  message: String
}

number: 6
description: buy a package with ID -> Called
route: https://localhost:5000/api/patient/buy-package/id=:id
method: POST
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
body:{
  {
    "products": [
        {
            "id": ObjectID,
            "quantity": Number,
        },
        {
            "id": ObjectID,
            "quantity": Number,
        }
    ]
}
}
returns: {
  message: String
  bill_id: ObjectID
}

number: 7
description: Delete a bill in case of Error
route: https://localhost:5000/api/patient/delete-bill/id=:id
method: DELETE
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
returns: {
  message: String
}

number: 8
description: Pay a bill
route: https://localhost:5000/api/patient/pay-bill/id=:id
method: POST
headers: {
  Bearer authorization: {
    type: String (JWT token)
  }
}
returns: {
  message: String
}


buyPackage flow:
buy package, call number 6 -> return bill ID -> {
if (login payment system successful) -> call number 8, number 8 sends a request to Payment system using axios -> {
  if (successful) -> save package usage history
  else (bug) -> number 8 catch error -> send error
  }
  else(login payment system fails) -> call number 7 delete bill 
}
